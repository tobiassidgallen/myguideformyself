Wallpaper Script

cat > ~/.config/scripts/wallpaper.sh <<'EOF'
#!/bin/bash

WALLPAPER_DIR="$HOME/Pictures/wallpapers"
FALLBACK_COLOR="#f9f5d7"  # Gruvbox Light background

# Validate image format
validate_image() {
    local file="$1"
    if command -v file >/dev/null 2>&1; then
        file_type=$(file --mime-type "$file" 2>/dev/null | cut -d: -f2 | tr -d ' ')
        case "$file_type" in
            image/jpeg|image/jpg|image/png|image/webp|image/bmp|image/gif|image/tiff)
                return 0
                ;;
            *)
                return 1
                ;;
        esac
    else
        # Fallback to extension checking if 'file' command is not available
        case "${file,,}" in
            *.jpg|*.jpeg|*.png|*.webp|*.bmp|*.gif|*.tiff|*.tif)
                return 0
                ;;
            *)
                return 1
                ;;
        esac
    fi
}

# Create wallpaper directory if it doesn't exist
if [ ! -d "$WALLPAPER_DIR" ]; then
    echo "Creating wallpaper directory: $WALLPAPER_DIR"
    mkdir -p "$WALLPAPER_DIR"
fi

# Search for valid image files
mapfile -d '' ALL_FILES < <(find "$WALLPAPER_DIR" -type f -readable -print0 2>/dev/null)

# Validate each file
WALLPAPERS=()
for file in "${ALL_FILES[@]}"; do
    if validate_image "$file"; then
        WALLPAPERS+=("$file")
    fi
done

if [ ${#WALLPAPERS[@]} -eq 0 ]; then
    echo "No valid wallpapers found in $WALLPAPER_DIR"
    echo "Available files:"
    find "$WALLPAPER_DIR" -type f -name "*.*" 2>/dev/null | head -5
    echo "Setting fallback solid color background..."

    # Check if feh is installed
    if command -v feh &> /dev/null; then
        feh --bg-fill --no-fehbg -B "$FALLBACK_COLOR"
        echo "Fallback background color set: $FALLBACK_COLOR"
    elif command -v xsetroot &> /dev/null; then
        xsetroot -solid "$FALLBACK_COLOR"
        echo "Fallback background color set with xsetroot: $FALLBACK_COLOR"
    else
        echo "Error: Neither feh nor xsetroot is available for setting background"
        exit 1
    fi
    exit 0
fi

echo "Found ${#WALLPAPERS[@]} valid wallpapers"

# Select random wallpaper
RANDOM_INDEX=$((RANDOM % ${#WALLPAPERS[@]}))
RANDOM_WALLPAPER="${WALLPAPERS[$RANDOM_INDEX]}"

# Validate selected file one more time
if [ ! -r "$RANDOM_WALLPAPER" ]; then
    echo "Error: Selected wallpaper is not readable: $RANDOM_WALLPAPER"
    exit 1
fi

# Check if feh is installed
if ! command -v feh &> /dev/null; then
    echo "Error: feh is not installed. Install it with: sudo pacman -S feh"
    exit 1
fi

# Set the wallpaper
if feh --bg-scale "$RANDOM_WALLPAPER" 2>/dev/null; then
    echo "Wallpaper set to: $(basename "$RANDOM_WALLPAPER")"
    echo "Full path: $RANDOM_WALLPAPER"

    # Optional: Save current wallpaper info
    echo "$RANDOM_WALLPAPER" > "$HOME/.config/scripts/current_wallpaper"
else
    echo "Error: Failed to set wallpaper with feh"
    echo "Falling back to solid color..."
    feh --bg-fill --no-fehbg -B "$FALLBACK_COLOR"
    exit 1
fi
EOF
chmod +x ~/.config/scripts/wallpaper.sh

i3lock Config

cat > ~/.config/scripts/lock_with_wallpaper.sh <<'EOF'
#!/bin/bash

# i3lock with blurred wallpaper - Gruvbox Light themed
WALLPAPER_FILE="$HOME/.config/scripts/current_wallpaper"
LOCK_IMAGE="/tmp/lock_wallpaper.png"
FALLBACK_COLOR="#fbf1c7"

# Check if we have a current wallpaper
if [ -f "$WALLPAPER_FILE" ]; then
    current_wall=$(cat "$WALLPAPER_FILE" 2>/dev/null)

    # Verify the wallpaper file exists and is readable
    if [ -f "$current_wall" ] && [ -r "$current_wall" ]; then
        echo "Creating blurred lock screen from: $current_wall"

        # Create blurred version with Gruvbox Light overlay
        convert "$current_wall" \
            -scale 1920x1080^ \
            -gravity center \
            -extent 1920x1080 \
            -blur 0x8 \
            -fill "#fbf1c7" \
            -colorize 15% \
            "$LOCK_IMAGE" 2>/dev/null

        # Check if conversion was successful
        if [ $? -eq 0 ] && [ -f "$LOCK_IMAGE" ]; then
            # Lock with the blurred wallpaper
            i3lock -i "$LOCK_IMAGE" \
                   --pointer=default \
                   --ignore-empty-password \
                   --show-failed-attempts

            # Clean up temporary file
            rm -f "$LOCK_IMAGE"
            exit 0
        else
            echo "Warning: Failed to create blurred wallpaper, using fallback"
        fi
    else
        echo "Warning: Wallpaper file not found or not readable: $current_wall"
    fi
else
    echo "Warning: No current wallpaper set"
fi

# Fallback to solid color
echo "Using fallback solid color lock screen"
i3lock -c "${FALLBACK_COLOR#\#}" \
       --pointer=default \
       --ignore-empty-password \
       --show-failed-attempts
EOF
